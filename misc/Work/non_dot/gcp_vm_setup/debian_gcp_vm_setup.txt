•CUDA setup
# Follow https://docs.nvidia.com/cuda/cuda-installation-guide-linux/
lspci | grep -i nvidia      // should return some ouptut denoting presence of a CUDA-capable gpu
hostnamectl                 // to check whether you have a supported version of linux
gcc --version               // gcc should be installed; if not installed, run ```sudo apt update && sudo apt install build-essential```
# Follow the CUDA Toolkit Installer (deb local) and Driver Installer (proprietary) instructions from https://developer.nvidia.com/cuda-downloads?target_os=Linux&target_arch=x86_64&Distribution=Debian&target_version=12&target_type=deb_local
sudo reboot
# Add the following line to .bashrc
export PATH=${PATH}:/usr/local/cuda-13.0/bin
